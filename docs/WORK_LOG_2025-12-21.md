```python
# AnalysisResultLog-оос түүхэн үр дүнг авах функц
def _get_qc_results_from_log(sample_ids: list, analysis_code: str = None):
    """Бүх түүхэн хэмжилтүүдийг авах (нэг дээж дээр олон удаа хэмжсэн)"""
    query = AnalysisResultLog.query.filter(
        AnalysisResultLog.sample_id.in_(sample_ids),
        AnalysisResultLog.final_result_snapshot.isnot(None)
```

```javascript
// ISO 80000-1 дугаарлалт: 5 бол дээш (0.695 → 0.70)
function roundHalfUp(num, decimals) {
  if (num == null || !Number.isFinite(num)) return null;
  const factor = Math.pow(10, decimals);
  return Math.round(num * factor + Number.EPSILON) / factor;
}
```

**Хэрэглэсэн газар:**
