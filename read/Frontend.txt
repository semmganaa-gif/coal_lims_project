<script>
let port; // Serial Port

// 1. ЖИНТЭЙ ХОЛБОГДОХ
document.getElementById('btnConnect').addEventListener('click', async () => {
    if ("serial" in navigator) {
        try {
            port = await navigator.serial.requestPort();
            await port.open({ baudRate: 9600, dataBits: 8, stopBits: 1, parity: "none" });
            
            document.getElementById('btnConnect').className = "btn btn-success";
            document.getElementById('btnConnect').innerText = "✅ Холбогдсон";
            logStatus("Жинтэй амжилттай холбогдлоо. Нүдэн дээр Enter дарж ашиглана уу.");
        } catch (err) {
            console.error(err);
            logStatus("Холбогдож чадсангүй. Гараар шивнэ үү.");
        }
    } else {
        alert("Таны хөтөч Web Serial API дэмжихгүй байна. Chrome ашиглана уу.");
    }
});

// 2. БҮХ ЖИНГИЙН НҮДЭНД EVENT LISTENER ОРУУЛАХ
document.querySelectorAll('.weight-input').forEach((input, index, allInputs) => {
    
    input.addEventListener('keydown', async (e) => {
        // ENTER дарах үед
        if (e.key === 'Enter') {
            e.preventDefault(); // Доошоо үсрэхийг зогсооно
            
            // Хэрэв жин холбогдоогүй бол гараар бичсэн гэж үзээд шууд дараагийн нүд рүү үсэрнэ
            if (!port) {
                focusNext(index, allInputs);
                return;
            }

            // Жин холбогдсон бол "Ухаалаг" горим ажиллана
            await fetchStableWeight(input);
            focusNext(index, allInputs);
        }
    });
});

// 3. ЖИН ТАТАХ ГОЛ ФУНКЦ (S команд)
async function fetchStableWeight(inputField) {
    try {
        logStatus("⏳ Жин тогтворжихийг хүлээж байна... (Гараа авна уу)");
        inputField.classList.add('running-input');
        inputField.value = ""; // Хуучин тоог арилгах

        // A. "S" (Send Stable) команд илгээх
        const encoder = new TextEncoder();
        const writer = port.writable.getWriter();
        await writer.write(encoder.encode("S\r\n")); 
        writer.releaseLock();

        // B. Хариу унших
        const reader = port.readable.getReader();
        const decoder = new TextDecoder();
        let buffer = "";

        while (true) {
            const { value, done } = await reader.read();
            if (done) break;
            if (value) {
                buffer += decoder.decode(value);
                // Mettler хариу: "S S    12.3456 g" (Stable)
                if (buffer.includes("\n")) {
                    const match = buffer.match(/(\d+\.\d+)/);
                    if (match) {
                        inputField.value = match[0];
                        logStatus("✅ Жин амжилттай орлоо: " + match[0] + "g");
                    } else {
                        logStatus("⚠️ Жингийн формат танигдсангүй.");
                    }
                    break; 
                }
            }
        }
        reader.releaseLock();
    } catch (err) {
        console.error(err);
        logStatus("❌ Алдаа гарлаа. Гараар бичнэ үү.");
    } finally {
        inputField.classList.remove('running-input');
    }
}

// 4. ТУСЛАХ ФУНКЦУУД
function focusNext(currentIndex, allInputs) {
    const nextInput = allInputs[currentIndex + 1];
    if (nextInput) {
        nextInput.focus();
    } else {
        // Хэрэв сүүлийн нүд бол түүхий өгөгдөл рүү үсрэх ч юм уу
        logStatus("Бүх дээж дууслаа.");
    }
}

function logStatus(msg) {
    document.getElementById('statusLog').innerText = "Төлөв: " + msg;
}

// Хадгалах товчны логик (API дуудах)
async function saveRow(btn) {
    const row = btn.closest('tr');
    const inputs = row.querySelectorAll('input');
    
    const data = {
        sample_id: inputs[0].value,
        mass: inputs[1].value,
        raw_value: inputs[2].value,
        equipment_id: document.getElementById('equipmentSelect').value
    };

    const res = await fetch('/save_result', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify(data)
    });
    const result = await res.json();
    if(result.success) {
        alert("Хариу: " + result.result.toFixed(2) + " J/g");
    }
}
